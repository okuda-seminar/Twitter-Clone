TABLE tweets {
  id UUID [pk]
  user_id UUID [not null]
  text varchar[140] [not null]
  created_at timestamp [default: 'now()']
}

TABLE likes {
  tweet_id INT [not null, ref: > tweets.id]
  user_id INT [not null]

  indexes {
    (tweet_id, user_id) [pk]
  }
}

TABLE retweets {
  tweet_id INT [not null, ref: > tweets.id]
  user_id INT [not null]

  indexes {
    (tweet_id, user_id) [pk]
  }
}
